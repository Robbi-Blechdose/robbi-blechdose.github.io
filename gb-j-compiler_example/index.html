GB-J Compiler: Example Program

2019-11-09

To demonstrate the basic concepts of the language, this example program was created.
It allows the user to move a cursor around on the screen and place tiles. Once finished, the result may be printed.

The file drawing_tiles.z80 is a tile data file generated with the tool GBTD.

import "stdlib/Utils.asm";
import "stdlib/Graphics/LCD.asm";
import "stdlib/Graphics/Tiles.asm";
import "stdlib/Graphics/Sprites.asm";
import "stdlib/GBPrinter.asm";
import "drawing_tiles.z80";

section Drawing
{
    package "ROM0";

    char x;
    char y;

    char tileID;

    Asm
    {
        //Use DW (B << 10) | (G << 5) | R
        //Palette 0
        "palettes::"
        "DW (31 << 10) | (31 << 5) | 31"
        "DW (20 << 10) | (20 << 5) | 20"
        "DW (10 << 10) | (10 << 5) | 10"
        "DW (0 << 10) | (0 << 5) | 0"
    }

    void main()
    {
        disableInterrupts();
        lcdOff();

        x = 8;
        y = 16;
        tileID = 2;

        //Allows for usage of sprites
        copyDMARoutine();
        //We don't want garbage onscreen
        clearOAM();
        
        //Get data from ASM references
        loadTiles(0, 0, 6, Asm.drawing_tiles);
        loadPalettes(0, Asm.palettes);

        //Cursor
        setSpriteTile(0, 1);
        //Indicator of current tile
        setSpriteTile(1, tileID);
        setSpritePosition(1, 155, 20);

        lcdOn(LCD_ON, WIN_MAP_9C00, WIN_OFF, BG_TILES_8000, BG_MAP_9800, OBJ8, OBJ_ON, BG_ON);
        enableInterrupts();

        //Main loop
        while(1 == 1)
        {
            readPad();

            if(isButtonPressed(BUTTON_RIGHT) == 1)
            {
                if(x <= 160)
                {
                    x = x + 1;
                }
            }
            else if(isButtonPressed(BUTTON_LEFT) == 1)
            {
                if(x > 8)
                {
                    x = x - 1;
                }
            }
            else if(isButtonPressed(BUTTON_UP) == 1)
            {
                if(y > 16)
                {
                    y = y - 1;
                }
            }
            else if(isButtonPressed(BUTTON_DOWN) == 1)
            {
                if(y < 152)
                {
                    y = y + 1;
                }
            }
            
            //Make sure the button is released
            if(isButtonPressed(BUTTON_B) == 0)
            {
                if(wasButtonPressed(BUTTON_B) == 1)
                {
                    tileID = tileID + 1;
                    if(tileID == 6)
                    {
                        tileID = 0;
                    }
                    else if(tileID == 1)
                    {
                        tileID = 2;
                    }
                    setSpriteTile(1, tileID);
                }
            }
            
            //Make sure the button is released
            if(isButtonPressed(BUTTON_START) == 0)
            {
                if(wasButtonPressed(BUTTON_START) == 1)
                {
                    disableInterrupts();
                    lcdOff();
                    loadWinMap(0, Asm._SCRN0, 0, 0, 32, 32);
                    lcdOn(LCD_ON, WIN_MAP_9C00, WIN_OFF, BG_TILES_8000, BG_MAP_9800, OBJ8, OBJ_ON, BG_ON);
                    enableInterrupts();
                    initPrinter();
                    transferData();
                    startPrint();
                }
            }

            setSpritePosition(0, x, y);

            halt();
        }
    }

    void VBlank()
    {
        char xTile;
        char yTile;

        dmaTransfer();

        if(isButtonPressed(BUTTON_A) == 1)
        {
            xTile = x / 8;
            xTile = xTile - 1;
            yTile = y / 8;
            yTile = yTile - 2;
            //Only modify VRAM in VBlank
            setBGTile(0, tileID, xTile, yTile);
        }
    }
}
